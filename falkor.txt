import os, subprocess, json, time, re
from pathlib import Path
from falkordb import FalkorDB
from dotenv import load_dotenv

# --- 1. SETUP E CONNESSIONE ---
load_dotenv()
os.environ["no_proxy"] = "localhost,127.0.0.1,padova.zucchettitest.it"

try:
    # Ci connettiamo al secondo container (Porta 6380)
    db = FalkorDB(host='localhost', port=6380)
    graph = db.select_graph('ZucchettiPureGraph')
    print("✅ Connesso al database PURE GRAPH (Porta 6380)")
except Exception as e:
    print(f"❌ Errore connessione: {e}")
    exit()

LLM_MODEL = "gemma3:27b-it-qat"

# --- 2. BRIDGE POWERSHELL ---
def _ps_invoke(url_suffix, payload):
    api_key = os.getenv("LITELLM_API_KEY")
    url = f"{os.getenv('LITELLM_BASE_URL').rstrip('/')}/{url_suffix}"
    stamp = str(time.time()).replace('.', '')
    req_file = Path(f"req_pure_{stamp}.json").absolute().as_posix()
    res_file = Path(f"res_pure_{stamp}.bin").absolute().as_posix()
    
    with open(req_file, "w", encoding="utf-8") as f:
        json.dump(payload, f, ensure_ascii=False)
    
    ps_cmd = f'$u="{url}"; $h=@{{"Authorization"="Bearer {api_key}";"Content-Type"="application/json"}}; try{{$b=[System.IO.File]::ReadAllBytes("{req_file}");$res=Invoke-WebRequest -Uri $u -Method Post -Headers $h -Body $b -Proxy $null -UseBasicParsing -TimeoutSec 120; [System.IO.File]::WriteAllBytes("{res_file}", $res.RawContentStream.ToArray())}} catch{{}}'
    
    subprocess.run(["powershell", "-Command", ps_cmd], capture_output=True)
    
    data = None
    if os.path.exists(res_file):
        with open(res_file, "rb") as f:
            try: data = json.loads(f.read().decode('utf-8-sig'))
            except: data = None
        os.remove(req_file); os.remove(res_file)
    return data

# --- 3. MOTORE DI RICERCA NEL GRAFO ---
def get_graph_context(query):
    """Estrae i fatti e i testi originali direttamente dal Grafo"""
    # Estrazione termini chiave dalla domanda
    words = re.findall(r'\b[A-Z][a-z]{2,}\b|\b[A-Z]{2,}\b|\b\w{6,}\b', query)
    entities = list(set([w.upper() for w in words if w.upper() not in ["DOMANDA", "ZUCCHETTI"]]))
    
    facts = []
    text_chunks = []
    
    print(f"   🔎 Ricerca relazionale per: {entities}")

    for ent in entities:
        # Questa query Cypher fa tutto: trova i fatti E il testo originale collegato
        q = f"""
        MATCH (c:Concept) WHERE c.name CONTAINS '{ent}'
        OPTIONAL MATCH (c)-[r]->(m:Concept)
        OPTIONAL MATCH (ch:Chunk)-[:CONTIENE]->(c)
        RETURN c.name, type(r), m.name, ch.content LIMIT 10
        """
        try:
            res = graph.query(q)
            for row in res.result_set:
                # Se c'è una relazione m-r-n, aggiungila ai fatti
                if row[1] and row[2]:
                    facts.append(f"- {row[0]} {row[1]} {row[2]}")
                # Se c'è del testo originale collegato, aggiungilo ai chunk
                if row[3]:
                    text_chunks.append(row[3])
        except:
            continue
            
    return list(set(facts)), list(set(text_chunks))

# --- 4. CICLO DI CHAT ---
def start_pure_chat():
    print(f"\n" + "="*60)
    print(f"🚀 ZUCCHETTI PURE GRAPHRAG (No Qdrant - Solo FalkorDB)")
    print("="*60 + "\n")
    
    while True:
        user_input = input("👤 Tu: ").strip()
        if user_input.lower() in ["esci", "quit"]: break
        if not user_input: continue

        start_time = time.time()

        # 1. Recupero dal Grafo
        facts, chunks = get_graph_context(user_input)
        
        # 2. Preparazione Contesto
        graph_txt = "\n".join(facts) if facts else "Nessuna relazione trovata nel grafo."
        # Limitiamo il testo originale per non appesantire il server
        docs_txt = "\n\n".join(chunks)[:5000] 

        # 3. Generazione Risposta
        sys_prompt = """Sei l'Assistente Pure-GraphRAG . 
        Tutte le tue informazioni provengono da un database a grafi.
        Usa le RELAZIONI per i fatti brevi e il TESTO CHUNK per spiegare i dettagli.
        Usa un linguaggio semplice e diretto.
        se trovi che una relazione è scritta con dei trattini, trasformala in una frase più naturale (es:DECISE_DI_PRENDERE -> "ha deciso di prendere ")
        se l'utente chiede una durata presente nella database , rispondi con la durata esatta, o rispondi con l'inizio e la fine della durata (es: durata della seconda guerra punica->dal 218 A.C. al 202 A.C. )" 
        Se le informazioni non sono nel contesto, rispondi 'info non esiste'."""
        
        user_prompt = f"""RELAZIONI ESTRATTE DAL GRAFO:
        {graph_txt}

        TESTO ORIGINALE DEI DOCUMENTI:
        {docs_txt}

        DOMANDA: {user_input}"""
        
        print("🧐 Navigazione dei nodi in corso...")
        
        res = _ps_invoke("chat/completions", {
            "model": LLM_MODEL, 
            "messages": [
                {"role": "system", "content": sys_prompt}, 
                {"role": "user", "content": user_prompt}
            ],
            "temperature": 0
        })
        
        elapsed = time.time() - start_time

        if res and 'choices' in res:
            print(f"\n🤖 Gemma3:\n{res['choices'][0]['message']['content']}")
            print(f"\n⏱️ Tempo di risposta: {elapsed:.2f}s")
        else:
            print("\n⚠️ Errore: Il server di Padova non ha risposto o il contesto era troppo grande.")

if __name__ == "__main__":
    start_pure_chat()
